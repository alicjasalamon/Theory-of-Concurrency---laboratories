-module(tw1c).

-export([start/0, aInit/0, bInit/0, cInit/0]).

start() -> 
	register(a, spawn(tw1c, aInit, [])),
	register(b, spawn(tw1c, bInit, [])),
	register(c, spawn(tw1c, cInit, [])),
	ok.
	
aInit() ->
	timer:sleep(1000),
	c ! aaa,
	aInit().

bInit() ->
	timer:sleep(1000),
	c ! bbb,
	bInit().
		
cInit() ->
	receive
		N -> io:format(N, " \n")
	end,
	cInit().
